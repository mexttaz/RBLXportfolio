BRivalsEngineCoreInput Buffering
-- M1 inputs can queue during recovery
local INPUT_BUFFER_WINDOW = 0.30
local inputBuffer = {}

if CombatService.IsInRecovery(player)
and inputType == "M1" then
    inputBuffer[player.UserId] = {
        inputType = inputType,
        payload = payload,
        standId = standId,
        time = tick(),
    }

    task.spawn(function()
        while inputBuffer[userId]
        and tick() - inputBuffer[userId].time
        < INPUT_BUFFER_WINDOW do
            if not CombatService.IsInRecovery(player) then
                local buffered = inputBuffer[userId]
                inputBuffer[userId] = nil

                BRivalsEngineCore.ProcessAbilityRequest(
                    player,
                    buffered.inputType,
                    buffered.payload
                )
                return
            end

            task.wait(0.016)
        end

        inputBuffer[userId] = nil
    end)
end

QueueStoreAdapterStudio-Safe Queueing
-- MemoryStore in live servers, fallback in Studio
local fallbackQueues = {}
local memoryStoreAvailable = true

function QueueStoreAdapter.UpsertTicket(ticket)
    local modeId = ticket.QueueMode
    local key = tostring(ticket.UserId)
    local ttl = CompetitiveConstants.Queue.TicketTtlSeconds

    if memoryStoreAvailable then
        local ok, err = pcall(function()
            local map = MemoryStoreService:GetSortedMap(
                getStoreKey(modeId)
            )

            map:SetAsync(
                key,
                ticket,
                ttl,
                ticket.Timestamp or os.time()
            )
        end)

        if ok then
            return true
        end

        memoryStoreAvailable = false
        warn("MemoryStore unavailable; using local fallback", err)
    end

    local queue = getFallbackQueue(modeId)
    queue[key] = {
        Value = ticket,
        ExpiresAt = os.clock() + ttl,
    }

    return true
end
MatchAcceptServiceAccept / Decline Flow
-- Match only advances when every participant accepts
local function onAcceptRemote(player, payload)
    if type(payload) ~= "table" then return end

    local matchToken = payload.MatchToken
    local action = payload.Action
    local match = MatchStateService.GetMatch(matchToken)

    if not match
    or match.State ~= CompetitiveConstants.MatchStates.Matching then
        return
    end

    if not isParticipant(match, player.UserId) then
        return
    end

    if os.clock() > match.AcceptDeadline then
        cancelMatch(matchToken, "TimedOut")
        return
    end

    if action == "Accept" then
        MatchStateService.MarkAccepted(matchToken, player.UserId)

        fireToMatchPlayers(match, "AcceptUpdate", {
            AcceptedCount = MatchAcceptService.GetAcceptedCount(matchToken),
            RequiredCount = #match.Players,
            RemainingSeconds = math.max(
                0,
                math.ceil(match.AcceptDeadline - os.clock())
            ),
        })

        if MatchStateService.HasAllAccepted(matchToken) then
            completeAccepted(matchToken)
        end
    elseif action == "Decline" then
        cancelMatch(matchToken, "Declined")
    end
end
MatchAcceptServiceTeleport Handoff
-- Once accepted, hand the match to teleport prep
local function completeAccepted(matchToken)
    local match = MatchStateService.GetMatch(matchToken)
    if not match then return end

    MatchStateService.SetMatchState(
        matchToken,
        CompetitiveConstants.MatchStates.Accepted
    )

    fireToMatchPlayers(match, "Accepted", {
        Message = "Teleport preparation starting",
        TeleportStub = false,
    })

    task.spawn(function()
        local ok, err =
            TeleportPreparationService.PrepareAndTeleport(match)

        if not ok then
            warn(
                "[MatchAcceptService] Teleport preparation failed",
                err
            )
        end
    end)
end
TeleportPreparationServiceReserved Server Teleport
-- Reserve server, build match packet, then teleport
function TeleportPreparationService.PrepareAndTeleport(match)
    local players = getMatchPlayers(match)
    if #players == 0 then
        return false, "NoPlayers"
    end

    if pendingTeleports[match.MatchToken] then
        return false, "TeleportAlreadyPending"
    end

    MatchStateService.SetMatchState(
        match.MatchToken,
        CompetitiveConstants.MatchStates.PreparingTeleport
    )

    for _, player in ipairs(players) do
        player:SetAttribute("CompetitiveMatchToken", match.MatchToken)
        player:SetAttribute("CompetitiveTeleportPending", true)
    end

    local reservedServerCode, reserveError =
        MatchReservationService.ReserveMatchServer(match.MatchToken)

    if not reservedServerCode then
        requeuePlayers(match, reserveError)
        return false, reserveError
    end

    local packet =
        MatchPacketService.CreateMatchPacket(match, reservedServerCode)

    local teleportData = TeleportDataSchema.Build({
        MatchPacket = packet,
        SourcePlaceId = game.PlaceId,
        SourceJobId = game.JobId,
    })

    return teleportAttempt(
        MatchReservationService.GetBetaMapPlaceId(),
        reservedServerCode,
        players,
        teleportData
    )
end
TeleportPreparationServiceFailure Recovery
-- Failed teleports push players back into recoverable lobby state
local function requeuePlayers(match, reason)
    for _, info in ipairs(match.Players or {}) do
        local player = Players:GetPlayerByUserId(info.UserId)

        MatchStateService.ClearPlayer(info.UserId)

        if player then
            LobbyPlayerStateService.CancelQueueScan(player)
            MatchTicketService.RemoveQueueTicket(player)
            MatchTicketService.RequeuePlayerIfStillInZone(
                player,
                match.ModeId
            )

            player:SetAttribute("CompetitiveTeleportPending", false)
            player:SetAttribute("CompetitiveTeleportFailed", true)
            player:SetAttribute(
                "CompetitiveTeleportReason",
                reason or "TeleportFailed"
            )

            TeleportEvent:FireClient(player, {
                State = "TeleportFailed",
                MatchToken = match.MatchToken,
                ModeId = match.ModeId,
                Reason = reason or "TeleportFailed",
            })
        end
    end

    MatchStateService.RemoveMatch(match.MatchToken)
end
RoundServiceRound Preparation
-- Reset players, freeze them, then run countdown
function RoundService.PrepareRound()
    Players.CharacterAutoLoads = false
    MVPService.ResetRound()

    resolving = false
    roundActive = false
    roundToken += 1

    local token = roundToken
    local _, _, completedRounds = ScoreService.GetScores()
    local upcoming = completedRounds + 1

    MatchSessionService.AssignSpawnPads()

    MatchStateService.SetRound(upcoming, {
        Type = "PrepareRound",
        RoundNumber = upcoming,
    })

    MatchStateService.SetState(defs.PreRound, {
        RoundNumber = upcoming,
    })

    for _, player in ipairs(expectedPlayers()) do
        if not player:GetAttribute("MatchForfeited") then
            resetPlayerForRound(player)
            hardPlaceAndFreeze(player, "PreRound")
        end
    end

    publishAlive()

    for i = rules.PreRoundCountdownSeconds, 1, -1 do
        UIEvent:FireAllClients({
            EventType = "UI",
            Type = "Countdown",
            Seconds = i,
            RoundNumber = upcoming,
        })

        task.wait(1)
        if token ~= roundToken or not loopStarted then return end
    end

    RoundService.StartRound(upcoming)
end
RoundServiceRound Resolution
-- One resolver owns score, overtime, match point, and victory
function RoundService.ResolveRoundEnd(winnerTeamId)
    if resolving or not roundActive then return end

    resolving = true
    roundActive = false

    MatchStateService.SetState(defs.RoundEnding, {
        WinnerTeamId = winnerTeamId,
    })

    for _, player in ipairs(expectedPlayers()) do
        player:SetAttribute("BRivals_RoundActive", false)
        player:SetAttribute("MatchAllowRoundRespawn", false)
    end

    FreezeService.FreezeTeam(expectedPlayers(), "RoundEnding")

    local team1, team2 = ScoreService.AddRoundWinner(winnerTeamId)
    local overtime = ScoreService.CheckOvertimeTransition()

    if overtime then
        OvertimeService.Start(overtime.Level, overtime.FirstTo)
    end

    if team1 == getModeRules().MatchPointAtWins
    or team2 == getModeRules().MatchPointAtWins then
        MatchPointService.Announce()
    end

    local winner = ScoreService.GetWinner()
    if winner then
        FreezeService.FreezeTeam(expectedPlayers(), "MatchEnding")
        VictoryService.EndMatch(winner, {
            Team1Score = team1,
            Team2Score = team2,
        })
    end
end
RoundServiceElimination Check
-- Round winner is determined from server-side alive state
local function aliveByTeam(teamId)
    local list = {}

    for _, player in ipairs(expectedPlayers()) do
        local sameTeam =
            TeamService.GetPlayerTeamId(player) == teamId

        local disqualified =
            player:GetAttribute("MatchDead")
            or player:GetAttribute("MatchForfeited")
            or player:GetAttribute("Match_SpectatorPlaceholder")

        if sameTeam and not disqualified then
            local hum = player.Character
                and player.Character:FindFirstChildOfClass("Humanoid")

            if hum and hum.Health > 0 then
                table.insert(list, player)
            end
        end
    end

    return list
end

function RoundService.CheckElimination()
    if not roundActive then return end

    local alive1 = #aliveByTeam(1)
    local alive2 = #aliveByTeam(2)

    UI.PublishAlive(alive1, alive2)

    if alive1 <= 0 and alive2 > 0 then
        RoundService.ResolveRoundEnd(2)
    elseif alive2 <= 0 and alive1 > 0 then
        RoundService.ResolveRoundEnd(1)
    end
end
MatchSessionServiceUnique Spawn Assignment
-- Spawn pads are assigned per team without duplicates
function MatchSessionService.AssignSpawnPads()
    if not activeSession then return end

    local requiredTeamCounts = { [1] = 0, [2] = 0 }

    for _, info in ipairs(activeSession.Players or {}) do
        local teamId = tonumber(info.TeamId) or 1
        info.TeamId = teamId
        requiredTeamCounts[teamId] =
            (requiredTeamCounts[teamId] or 0) + 1
    end

    local selectedMapId = SpawnMapService.SelectMapId(activeSession)
    local teamPads = {
        [1] = SpawnMapService.GetUniqueTeamPads(
            activeSession.ModeId, 1,
            selectedMapId, requiredTeamCounts[1]
        ),
        [2] = SpawnMapService.GetUniqueTeamPads(
            activeSession.ModeId, 2,
            selectedMapId, requiredTeamCounts[2]
        ),
    }

    table.clear(activeSession.SpawnAssignments)

    local teamIndexes = { [1] = 0, [2] = 0 }
    local usedPads = {}

    for _, info in ipairs(activeSession.Players or {}) do
        local teamId = tonumber(info.TeamId) or 1
        teamIndexes[teamId] += 1

        local spawnPart = teamPads[teamId][teamIndexes[teamId]]

        if spawnPart and not usedPads[spawnPart] then
            usedPads[spawnPart] = true
            activeSession.SpawnAssignments[info.UserId] = spawnPart
        end
    end

    activeSession.SpawnAssignmentReady = true
end
RewardServiceIdempotent Rewards
-- RewardHistory prevents double-awarding the same match
local function persistReward(userId, reward)
    local ok, err

    for attempt = 1, 5 do
        ok, err = pcall(function()
            profileStore:UpdateAsync(tostring(userId), function(old)
                old = type(old) == "table" and old or {}

                old.RewardHistory =
                    type(old.RewardHistory) == "table"
                    and old.RewardHistory
                    or {}

                local key = tostring(reward.MatchToken)
                    .. ":"
                    .. tostring(userId)

                if old.RewardHistory[key] then
                    return old
                end

                old.RewardHistory[key] = os.time()
                old.Coins += reward.Coins
                old.Elo = math.clamp(
                    old.Elo + reward.EloDelta,
                    0,
                    100000
                )

                return old
            end)
        end)

        if ok then return true end
        task.wait(math.min(0.8 * attempt, 4))
    end

    warn("[RewardService] Persist failed", userId, err)
    return false
end
RewardServiceElo Delta Formula
-- Elo adjusts based on expected win probability
local function eloDelta(myAvg, oppAvg, won)
    myAvg = math.clamp(
        tonumber(myAvg) or 1000, 0, 100000
    )
    oppAvg = math.clamp(
        tonumber(oppAvg) or 1000, 0, 100000
    )

    local expected =
        1 / (1 + 10 ^ ((oppAvg - myAvg) / 400))

    local score = won and 1 or 0
    local raw = 60 * (score - expected)

    local delta = math.floor(
        raw + (raw >= 0 and 0.5 or -0.5)
    )

    if won then
        return math.clamp(delta, 20, 100)
    end

    return math.clamp(delta, -100, -20)
end
StandServiceUnified Roll Categories
-- One gacha pipeline can roll cosmetics, stands, perks, and pets
local ROLL_CATEGORIES = {
    { id = "Cosmetic", weight = 75 },
    { id = "Stand",    weight = 5 },
    { id = "Perk",     weight = 20 },
    { id = "Pet",      weight = 0 },
}

local function pickCategory()
    local totalWeight = 0

    for _, cat in ipairs(ROLL_CATEGORIES) do
        totalWeight += cat.weight
    end

    if totalWeight <= 0 then
        return "Stand"
    end

    local roll = math.random() * totalWeight
    local cumulative = 0

    for _, cat in ipairs(ROLL_CATEGORIES) do
        cumulative += cat.weight

        if roll <= cumulative then
            return cat.id
        end
    end

    return "Stand"
end

-- Mobile inputs respect menus, lobby state, cooldowns, and combat locks
local function requestAction(action)
    if editMode
    or isBlocking
    or not ACTION_SET[action]
    or menuLocked()
    or lobbyBlocked() then
        return
    end

    local character = player.Character

    if player:GetAttribute("_IsGrabbed")
    or (character and character:GetAttribute("_IsGrabbed")) then
        InputEvent:FireServer("GrabEscape")
        return
    end

    if player:GetAttribute("CombatLocked")
    or player:GetAttribute("CannotAttack") then
        return
    end

    local stand = player:GetAttribute("EquippedStand")
    if not stand or stand == "" then return end

    if cooldownActive(stand, action) then
        if _G.ShowCooldownBlockedCue then
            _G.ShowCooldownBlockedCue({ Duration = 0.22 })
        end
        return
    end

    notifyTutorialMobileAction(action, true)
    InputEvent:FireServer(action, aimPayload())
end

-- Rebuilds the ability HUD if it goes missing or stale
local HUD_NAME = "StandHUDGui"
local CHECK_INTERVAL = 1.5
local STALE_AFTER = 6

local function cloneFresh(reason)
    local existing = playerGui:FindFirstChild(HUD_NAME)
    if existing then existing:Destroy() end

    local template = StarterGui:FindFirstChild(HUD_NAME)

    if not template then
        warn("[StandHUDWatchdog] Missing HUD template", reason)
        return nil
    end

    local clone = template:Clone()
    clone.ResetOnSpawn = false
    clone.Enabled = true
    clone.Parent = playerGui

    _G.BRivalsHUDAlive = tick()

    warn("[StandHUDWatchdog] Rebuilt Stand HUD", reason)
    return clone
end

local function ensureHud(reason)
    local hud = playerGui:FindFirstChild(HUD_NAME)

    if not hud then
        return cloneFresh(reason or "missing")
    end

    local hasPanels =
        hud:FindFirstChild("ControlsPanel") ~= nil
        and hud:FindFirstChild("CooldownsPanel") ~= nil

    local alive = tonumber(_G.BRivalsHUDAlive) or 0

    if not hasPanels and tick() - alive > STALE_AFTER then
        return cloneFresh(reason or "controller-stale")
    end

    return hud
end

Engineering Decisions

The details that make the work usable on a team.

A hiring page should not just say "I made systems." It should show how I think: clear authority boundaries, reusable services, predictable module structure, testable infrastructure, and client code that supports real Roblox players on real devices.

Client responsiveness without client-owned damage

Inputs can feel immediate while damage, stun, block, parry, hitbox, and recovery outcomes remain server-owned.

Character kits are content, not architecture rewrites

New stands follow a controller/state/ability structure so the roster can grow without duplicating core combat logic.

Lobby and match servers have different responsibilities

The main place prepares players through queueing, progression, and onboarding while Competitive Alpha handles match runtime.

Match flow accounts for failure cases

Reconnect, AFK, forfeit, spectator, return-to-lobby, cleanup, and reward services exist because real multiplayer sessions break.

Networking is grouped by gameplay domain

Combat, progression, inventory, matchmaking, party flow, competitive state, score, round, spectator, and reward events are separated by purpose.

Studio-safe fallbacks keep development moving

Queue storage can fall back locally when MemoryStore is unavailable, which keeps matchmaking systems testable during Studio development.

Progression uses registries and appliers

Stands, cosmetics, pets, perks, rewards, and rig application are represented through modules instead of being buried in UI scripts.

Client UX includes recovery paths

Mobile controls, HUD watchdogs, resolution scaling, spectator UI, reward presentation, and VFX controllers support real player conditions.
